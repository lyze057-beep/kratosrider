#### 2. 添加位置上报业务逻辑

```go
// internal/biz/location.go
type LocationUsecase struct {
    log          *log.Helper
    locationRepo LocationRepo
    rdb          *redis.Client
}

// UpdateLocation 更新骑手位置
func (uc *LocationUsecase) UpdateLocation(ctx context.Context, riderID int64, lat float64, lng float64) error {
    // 1. 验证骑手是否存在
    rider, err := uc.locationRepo.GetRiderByID(ctx, riderID)
    if err != nil {
        return err
    }
    
    // 2. 百度地图坐标转换（如果是GPS坐标）
    // lat, lng = uc.convertGPSToBaidu(lat, lng)
    
    // 3. 保存位置数据
    location := &model.Location{
        RiderID:  riderID,
        Latitude: lat,
        Longitude: lng,
        Accuracy: 10.0,
        Speed:    0,
        Direction: 0,
        Address:  "",
        City:     "北京",
        Province: "北京",
        Country:  "中国",
    }
    
    err = uc.locationRepo.SaveLocation(ctx, location)
    if err != nil {
        return err
    }
    
    // 4. 更新Redis缓存（5分钟过期）
    cacheKey := fmt.Sprintf("rider:location:%d", riderID)
    data, _ := json.Marshal(location)
    uc.rdb.Set(ctx, cacheKey, data, 5*time.Minute)
    
    // 5. 推送位置给订单系统（用于智能派单）
    if uc.orderMessageProducer != nil {
        go func() {
            msg := &data.LocationMessage{
                RiderID: riderID,
                Lat:     lat,
                Lng:     lng,
            }
            uc.orderMessageProducer.PublishLocationMessage(ctx, msg)
        }()
    }
    
    return nil
}
```

#### 3. 添加位置上报HTTP服务

```go
// internal/service/location.go
type LocationService struct {
    v1.UnimplementedLocationServer
    locationUsecase biz.LocationUsecase
    log             *log.Helper
}

func (s *LocationService) UpdateLocation(ctx context.Context, req *v1.UpdateLocationRequest) (*v1.UpdateLocationReply, error) {
    err := s.locationUsecase.UpdateLocation(ctx, req.RiderId, req.Latitude, req.Longitude)
    if err != nil {
        s.log.Error("update location failed", "err", err)
        return &v1.UpdateLocationReply{
            Success: false,
            Message: err.Error(),
        }, nil
    }
    
    return &v1.UpdateLocationReply{
        Success: true,
        Message: "位置上报成功",
    }, nil
}
```
